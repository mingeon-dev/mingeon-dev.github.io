- 연달아 호출되는 함수의 마지막만 호출하도록 하는 것
- window resize, 검색 등과 같이 연속된 이벤트가 발생하는 경우 사용

```javascript
var timer;
// 이벤트 리스너 내부
if (timer) {
  clearTimeout(timer);
}
timer = setTimeout(function () {
  // 로직
}, 200);
```
